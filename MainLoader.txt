    if UserInputService.TouchEnabled and not UserInputService.KeyboardEnabled then

        return "PC"

local function tween(obj, props, duration, style, direction) 
    if not obj or not obj.Parent then return end 
    pcall(function() 
        TweenService:Create(
            obj, 
            TweenInfo.new(duration or 0.3, style or Enum.EasingStyle.Quad, direction or Enum.EasingDirection.Out), 
            props
        ):Play() 
    end) 
end

local function createMenuButton(parent, text, code) 
    local btn = Instance.new("TextButton") 
    btn.Size = UDim2.new(1,-10,0,40) 
    btn.Position = UDim2.new(0,5,0,#parent:GetChildren()*45) 
    btn.BackgroundColor3 = COLORS.Frame 
    btn.Text = text 
    btn.TextColor3 = COLORS.White 
    btn.Font = Enum.Font.GothamBold 
    btn.TextSize = 12 
    btn.BorderSizePixel = 0 
    round(btn,8) 
    btn.Parent = parent

    btn.MouseEnter:Connect(function() 
        tween(btn, {BackgroundColor3 = COLORS.DarkBG}, 0.2) 
    end)

    btn.MouseLeave:Connect(function() 
        tween(btn, {BackgroundColor3 = COLORS.Frame}, 0.2) 
    end)

    btn.MouseButton1Click:Connect(function() 
        notif("Executado", "Executado: "..text) 
        pcall(function() loadstring(code)() end) 
    end)

    return btn 
end
